// dp[n][x] + (n-x <= x?) + sum(dp[n-x][1 -> min(x,n-x-1)])


fn main() {
    let mut dp = Vec::new();
    let mut dpsum = Vec::new();
    dp.push(Vec::new());
    dpsum.push(Vec::new());
    let mut i = 1;
    loop{
        dp.push(Vec::new());
        dpsum.push(Vec::new());
        dp[i].push(0);
        dpsum[i].push(0);
        let mut cnt = 0;
        for j in 1..i+1 {
            dp[i].push(0);
            dpsum[i].push(0);
            if i == j {
                dp[i][j] = 1;
            }else{
                dp[i][j] = dpsum[i-j][cmp::min(j,i-j)];
            }
            cnt += dp[i][j];
            cnt %= 1_000_000;
            dpsum[i][j] = (dpsum[i][j-1]+dp[i][j]) % 1_000_000;
        }
        println!("{}=>{}",i,cnt);
        if cnt % 1_000_000 == 0 {
            println!("{}=>{}",i,cnt);
            return ;
        }
        i+=1;
    }
}
