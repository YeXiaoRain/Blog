##各个执行块

<judge block> ::= <if statement>

_func _void quickSort{
    _para{
        _int [] s;
        _int l
        _int r;
    };
    _var{
        _int i;
    };
    _pro{
        _if (l< r){
            i = l;
            j = r;
            x = s[l];
            _while (i < j){
                _while(i < j _and s[j]>= x){
                    j=j-1;
                };
                _if(i < j){
                    s[i] = s[j];
                    i=i+1;
                }{};
                _while(i < j _and s[i]< x){
                    i=i+1;
                };
                _if(i < j){
                    s[j] = s[i];
                    j=j-1;
                }{};
            };
            s[i] = x;
            quickSort(s, l, i - 1); 
            quickSort(s, i + 1, r);
        }{};
    };
};
